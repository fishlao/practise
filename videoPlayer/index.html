这是一个简单的播放器插件
		使用方法
		new MusicPlayer({
			audioUrl: [
				{ title: 'heart', audioUrl: "audio/heart.mp3" },
				{ title: 'seeU', audioUrl: "audio/seeU.mp3" },
			], //视频播放地址，[{title: xxx, audioUrl: xxx}] / {title:xxx, audioUrl: xxxx}
			nodeId: 'container', //音乐播放器放置的容器,
			iconSkin: '', //播放器按钮背景图
			boxStyle: 'background-color: rgba(128, 128, 128, 0.31)', //播放器的样式
			autoPlay: true, // 是否自动播放
			loop: false, // 是否循环播放
		})
